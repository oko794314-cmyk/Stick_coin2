МИТТЄВА КНОПКА START/STOP const mbtn = document.getElementById('mbtn'); mbtn.onclick = function() { if (!isM) { startMining(); } else { stopMining(); } }; function startMining() { isM = true; mbtn.innerText = "STOP"; mbtn.classList.add('active'); localStorage.setItem('mine_active_' + cur, 'true'); db.users[cur].active_nodes = (db.users[cur].active_nodes || 0) + 1; loop(); sync(); // Одразу пінгуємо сервер про запуск } function stopMining() { isM = false; mbtn.innerText = "START"; mbtn.classList.remove('active'); localStorage.setItem('mine_active_' + cur, 'false'); db.users[cur].active_nodes = Math.max(0, (db.users[cur].active_nodes || 1) - 1); sync(); // Одразу зберігаємо прогрес } function loop() { if (isM) { loc += BASE_RATE; upd(); setTimeout(loop, 1000); } } function upd() { document.getElementById('bal').innerText = (cloud + loc).toFixed(5); document.getElementById('stat-bal').innerText = (cloud + loc).toFixed(2); document.getElementById('active-nodes').innerText = db.users[cur].active_nodes || 0; } async function sync() { if (!cur) return; db.users[cur].balance = cloud + loc; await fetch(`https://api.jsonbin.io/v3/b/${BID}`, { method: "PUT", headers: {"X-Master-Key": KEY, "Content-Type": "application/json"}, body: JSON.stringify(db) }); cloud += loc; loc = 0; } }; // Функція виходу, яка очищує пам'ять function logout() { localStorage.clear(); location.reload(); }
